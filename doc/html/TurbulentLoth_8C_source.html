74 Foam::tmp<Foam::volScalarField>
75 Foam::particleDragModels::TurbulentLoth::turbulentDispersion() const
76 {
77  // Direction dependent parameter
78  const volVectorField& Ur(pair_.Ur());
79  const volScalarField& magUr(pair_.magUr());
80  const volVectorField& Up(pair_.dispersed().U());
81  const volScalarField& magUp(mag(Up));
82  const volScalarField Cbeta(1.8 - 1.35*sqr((Ur&Up)/(magUr*magUp)));
83 
84  // Time scale of fluid turbulent motion
85  const volScalarField Kpg
86  (
87  this->db().lookupObject<volScalarField>("Kpg." + pair_.dispersed().name())
88  );
89 
90  const volScalarField Kg
91  (
92  this->db().lookupObject<volScalarField>("k." + pair_.continuous().name())
93  );
94 
95  const volScalarField epsilonG
96  (
97  this->db().lookupObject<volScalarField>("epsilon." + pair_.continuous().name())
98  );
99 
100  const volScalarField zetar(3.0*sqr(magUr)/(2.0*Kg));
101  const volScalarField taupg((Kg/epsilonG)/sqrt(1.0 + Cbeta*zetar));
102 
103  return (1.0/3.0)*taupg*Kpg;
104 }

106 Foam::tmp<Foam::volVectorField>
107 Foam::particleDragModels::TurbulentLoth::Ur() const
108 {
109  //- Calculate Turbulent Dispersion Coefficient and drifting velocity
110  const tmp<volScalarField> tDpg(turbulentDispersion());
111  const volScalarField& Dpg(tDpg());
112 
113  const volScalarField& alphaP(pair_.dispersed());
114  const volScalarField& alphaG(pair_.continuous());
115  const volVectorField Ud(-Dpg*((1.0/alphaP)*fvc::grad(alphaP) - (1.0/alphaG)*fvc::grad(alphaG)));
116 
117  return pair_.Ur() - Ud;
118 }

126 Foam::scalar Foam::particleDragModels::TurbulentLoth::JM
127 (const scalar& Ma) const
128 {
129  return
130  neg0(Ma - 1)*(2.26 - 0.1/max(Ma, SMALL) + 0.14/max(pow(Ma, 3), SMALL))
131  + pos(Ma - 1)*(1.6 + 0.25/max(Ma, SMALL) + 0.11/max(sqr(Ma), SMALL) + 0.44/max(pow(Ma, 3), SMALL));

134 Foam::scalar Foam::particleDragModels::TurbulentLoth::Cdfm
135 (const scalar& S) const
136 {
137  return
138  (1 + 2*sqr(S))*exp(-sqr(S))/max((sqrt(constant::mathematical::pi)*pow(S, 3)), SMALL)
139  + 2*sqrt(constant::mathematical::pi)/max((3*S), SMALL)
140  + (4*pow(S, 4) + 4*sqr(S) - 1)*std::erf(S)/max((2*pow(S, 4)), SMALL);
141 }

179  neg0(Ma - 0.8)*(166*pow(Ma, 3) + 3.29*sqr(Ma) - 10.9*Ma + 20)

191 Foam::scalar Foam::particleDragModels::TurbulentLoth::CdComp
192 (const scalar& Re, const scalar& Ma) const
193 {
194  return
195  24*(1 + 0.15*pow(Re, 0.687))*HM(Ma)
196  + Re*0.42*CM(Ma)/(1 + 42500/max(pow(Re, 1.16*CM(Ma)), SMALL)
197  + GM(Ma)/max(sqrt(Re), SMALL));

200 Foam::tmp<Foam::volScalarField> Foam::particleDragModels::TurbulentLoth::CdRe() const
201 {
202  const tmp<volScalarField> tT(pair_.continuous().thermo().T());
203  const volScalarField& T(tT());
204 
205  const tmp<volScalarField> tdp(pair_.dispersed().d());
206  const volScalarField& dp(tdp());
207 
208  const tmp<volScalarField> tmu(pair_.continuous().nu()*pair_.continuous().rho());
209  volScalarField mug(tmu());
210 
211  //- Calculate Turbulent Dispersion Coefficient and drifting velocity
212  const tmp<volScalarField> tDpg(turbulentDispersion());
213  const volScalarField& Dpg(tDpg());
214 
215  const volScalarField& alphaP(pair_.dispersed());
216  const volScalarField& alphaG(pair_.continuous());
217  const volVectorField Ud(-Dpg*((1.0/alphaP)*fvc::grad(alphaP) - (1.0/alphaG)*fvc::grad(alphaG)));
218 
219  const tmp<volVectorField> tUr(pair_.Ur() - Ud);
220 
221  const tmp<volScalarField> tMagUr(mag(tUr()));
222  const volScalarField& magUr(tMagUr());
223 
224  //- Calculate Instantaneous relative velocity
225  const volScalarField Kpg
226  (
227  this->db().lookupObject<volScalarField>("Kpg." + pair_.dispersed().name())
228  );
229  const volScalarField Kg
230  (
231  this->db().lookupObject<volScalarField>("k." + pair_.continuous().name())
232  );
233  const volScalarField Theta
234  (
235  this->db().lookupObject<volScalarField>("Theta." + pair_.dispersed().name())
236  );
237 
238  const tmp<volScalarField> tmagUri(sqrt(sqr(magUr) + (2.0/3.0)*(Kg - Kpg) + Theta));
239  const volScalarField& magUri(tmagUri());
240 
241  // Loth Drag Model
242  const tmp<volScalarField> tM(max(magUri/sqrt(gamma_*R_*T), SMALL));
243  const volScalarField& M(tM());
244 
245  const tmp<volScalarField> tRe(max(pair_.continuous().rho()*magUri*dp/mug, SMALL));
246  const volScalarField& Re(tRe());
247 
248  const tmp<volScalarField> tS(sqrt(gamma_/2)*M);
249  const volScalarField& S(tS());
250 
251  const tmp<volScalarField> tKn(sqrt(constant::mathematical::pi)*S/Re);
252  const volScalarField& Kn(tKn());
253 
254  volScalarField CdRe_
255  (
256  IOobject
257  (
258  "CdRe",
259  pair_.phase1().mesh()
260  ),
261  pair_.phase1().mesh(),
262  dimensionedScalar("", dimless, 0.0)
263  );
264 
265  forAll(Re, i)
266  {
267  CdRe_[i] =
268  (
269  Re[i] <= 45 ? CdRare(Re[i], M[i], Kn[i], S[i])
270  : CdComp(Re[i], M[i])
271  );
272  }
273 
274  // Update Boundary patches
275  volScalarField::Boundary& CdReB(CdRe_.boundaryFieldRef());
276  const volScalarField::Boundary& MB(M.boundaryField());
277  const volScalarField::Boundary& ReB(Re.boundaryField());
278  const volScalarField::Boundary& SB(S.boundaryField());
279  const volScalarField::Boundary& KnB(Kn.boundaryField());
280 
281  forAll(CdReB, k)
282  {
283  Field<scalar>& pF(CdReB[k]);
284  const Field<scalar>& MpF(MB[k]);
285  const Field<scalar>& RepF(ReB[k]);
286  const Field<scalar>& SpF(SB[k]);
287  const Field<scalar>& KnpF(KnB[k]);
288 
289  forAll(pF, i)
290  {
291  pF[i] =
292  (
293  RepF[i] <= 45 ? CdRare(RepF[i], MpF[i], KnpF[i], SpF[i])
294  : CdComp(RepF[i], MpF[i])
295  );
296  }
297  }
298  // CdRe_.correctBoundaryConditions();
299 
300  return Foam::tmp<Foam::volScalarField>(new volScalarField ("tCdRe", CdRe_));
301 }
